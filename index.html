  if(mode > 0) {
    unsigned long now = millis();
    
    switch(mode) {
      case 1:
        if(now - lastModeUpdate > 20) {
          modeStep = (modeStep + 1) % 1536;
          int r, g, b;
          if(modeStep < 256) { r = 1023; g = modeStep * 4; b = 0; }
          else if(modeStep < 512) { r = 1023 - (modeStep - 256) * 4; g = 1023; b = 0; }
          else if(modeStep < 768) { r = 0; g = 1023; b = (modeStep - 512) * 4; }
          else if(modeStep < 1024) { r = 0; g = 1023 - (modeStep - 768) * 4; b = 1023; }
          else if(modeStep < 1280) { r = (modeStep - 1024) * 4; g = 0; b = 1023; }
          else { r = 1023; g = 0; b = 1023 - (modeStep - 1280) * 4; }
          analogWrite(redPin, r);
          analogWrite(greenPin, g);
          analogWrite(bluePin, b);
          lastModeUpdate = now;
        }
        break;
      case 2:
        if(now - lastModeUpdate > 100) {
          int f = random(600, 1023);
          analogWrite(redPin, f);
          analogWrite(greenPin, f/8);
          analogWrite(bluePin, 0);
          lastModeUpdate = now;
        }
        break;
      case 3:
        if(now - lastModeUpdate > 50) {
          modeStep = (modeStep + 5) % 1024;
          float sv = (sin(modeStep * 0.006) * 511) + 512;
          analogWrite(redPin, (modeRed * (int)sv) / 1023);
          analogWrite(greenPin, (modeGreen * (int)sv) / 1023);
          analogWrite(bluePin, (modeBlue * (int)sv) / 1023);
          lastModeUpdate = now;
        }
        break;
      case 4:
        if(now - lastModeUpdate > 100) {
          if(modeStep == 0) {
            analogWrite(redPin, modeRed);
            analogWrite(greenPin, modeGreen);
            analogWrite(bluePin, modeBlue);
            modeStep = 1;
          } else {
            analogWrite(redPin, 0);
            analogWrite(greenPin, 0);
            analogWrite(bluePin, 0);
            modeStep = 0;
          }
          lastModeUpdate = now;
        }
        break;
      case 5:
        if(now - lastModeUpdate > 200) {
          if(random(0, 3) == 0) {
            analogWrite(redPin, modeRed);
            analogWrite(greenPin, modeGreen);
            analogWrite(bluePin, modeBlue);
          } else {
            analogWrite(redPin, modeRed/3);
            analogWrite(greenPin, modeGreen/3);
            analogWrite(bluePin, modeBlue/3);
          }
          lastModeUpdate = now;
        }
        break;
      case 6:
        if(now - lastModeUpdate > 20) {
          modeStep = (modeStep + 2) % 1024;
          float pv = (sin(modeStep * 0.006) * 511) + 512;
          analogWrite(redPin, (modeRed * (int)pv) / 1023);
          analogWrite(greenPin, (modeGreen * (int)pv) / 1023);
          analogWrite(bluePin, (modeBlue * (int)pv) / 1023);
          lastModeUpdate = now;

        break;

        if(now - lastModeUpdate > 30) {

          analogWrite(greenPin, g);
