📟 Код:

#include <ESP8266WiFi.h> #include <DHT.h> const char* ssid = "ваш Wi-Fi"; const char* password = "ваш пароль"; DHT dht(2, DHT22); void setup() { Serial.begin(115200); dht.begin(); WiFi.begin(ssid, password); } void loop() { float t = dht.readTemperature(); float h = dht.readHumidity(); Serial.print("Temp: "); Serial.println(t); delay(2000); }
