📟 ПОЛНЫЙ КОД МЕТЕОСТАНЦИИ С ДИСПЛЕЕМ

#include <ESP8266WiFi.h>
#include <WiFiClientSecure.h>
#include <UniversalTelegramBot.h>
#include <DHT.h>
#include <LiquidCrystal_I2C.h>

// ========== ТВОИ ДАННЫЕ ==========
const char* ssid = "ваш Wi-Fi";
const char* password = "ваш пароль";
#define BOT_TOKEN "токен_бота"

// ID разрешенных пользователей
const String allowedUsers[] = {
  "1234567890",
  "123456789",
  "1234567890"
};
const int totalUsers = 3;
// =================================

// Пины
#define DHTPIN 0        // D3 (GPIO0)
#define DHTTYPE DHT22
#define I2C_SDA 4       // D2
#define I2C_SCL 5       // D1

DHT dht(DHTPIN, DHTTYPE);
WiFiClientSecure client;
UniversalTelegramBot bot(BOT_TOKEN, client);
LiquidCrystal_I2C lcd(0x27, 16, 2);  // Адрес I2C может быть 0x27 или 0x3F

float temperature = 0;
float humidity = 0;
unsigned long lastTime = 0;

bool isAllowed(String chat_id) {
  for(int i = 0; i < totalUsers; i++) {
    if(allowedUsers[i] == chat_id) return true;
  }
  return false;
}

void setup() {
  Serial.begin(115200);
  
  // Инициализация дисплея
  Wire.begin(I2C_SDA, I2C_SCL);
  lcd.init();
  lcd.backlight();
  lcd.setCursor(0, 0);
  lcd.print("Метеостанция");
  lcd.setCursor(0, 1);
  lcd.print("Запуск...");
  
  dht.begin();
  client.setInsecure();
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("WiFi OK");
  lcd.setCursor(0, 1);
  lcd.print(WiFi.localIP().toString());
  delay(2000);
  
  bot.sendMessage("404415869", "꧁༺ *БОТ ЗАПУЩЕН* ༻꧂", "Markdown");
}

void loop() {
  if(millis() - lastTime > 5000) {
    float t = dht.readTemperature();
    float h = dht.readHumidity();
    
    if(!isnan(t) && !isnan(h)) {
      temperature = t;
      humidity = h;
      
      // Обновление дисплея
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Temp: ");
      lcd.print(temperature, 1);
      lcd.print(" C");
      
      lcd.setCursor(0, 1);
      lcd.print("Hum:  ");
      lcd.print(humidity, 1);
      lcd.print(" %");
    }
    
    int msgCount = bot.getUpdates(bot.last_message_received + 1);
    
    for(int i = 0; i < msgCount; i++) {
      String chat = bot.messages[i].chat_id;
      String text = bot.messages[i].text;
      String name = bot.messages[i].from_name;
      
      if(!isAllowed(chat)) {
        bot.sendMessage(chat, "⛔️ *НЕТ ДОСТУПА*", "Markdown");
        continue;
      }
      
      if(text == "/start") {
        String msg = "꧁༺ *МЕТЕОСТАНЦИЯ* ༻꧂\n\n";
        msg += "*ПРИВЕТ, " + name + "*\n";
        msg += "🌡 /temp - температура\n";
        msg += "💧 /hum - влажность\n";
        msg += "📊 /all - все данные";
        bot.sendMessage(chat, msg, "Markdown");
      }
      
      else if(text == "/temp") {
        String msg = "🌡 *Температура:* " + String(temperature, 1) + "°C";
        bot.sendMessage(chat, msg, "Markdown");
      }
      
      else if(text == "/hum") {
        String msg = "💧 *Влажность:* " + String(humidity, 1) + "%";
        bot.sendMessage(chat, msg, "Markdown");
      }
      
      else if(text == "/all") {
        String msg = "📊 *Данные метеостанции*\n\n";
        msg += "🌡 Температура: " + String(temperature, 1) + "°C\n";
        msg += "💧 Влажность: " + String(humidity, 1) + "%";
        bot.sendMessage(chat, msg, "Markdown");
      }
    }
    
    lastTime = millis();
  }
}
